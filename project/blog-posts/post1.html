Python Code Example: Carbon Emission Reduction

            
# Energy produced by solar panels (in kWh per day)
solar_energy_per_day = 30  

# CO2 emissions per kWh for coal (in kg)
coal_emission_per_kWh = 0.91  

# Calculate CO2 saved per year
co2_saved_per_year = solar_energy_per_day * coal_emission_per_kWh * 365

print(f"Using solar energy instead of coal can reduce CO2 emissions by {co2_saved_per_year:.2f} kg per year.")
